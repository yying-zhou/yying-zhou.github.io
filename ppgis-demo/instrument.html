1 — load proxy mesh (real implementation)
import { GLTFLoader } from 'three/addons';
// Luma AI exports .splat + .glb together
loader.load('park_proxy.glb', (gltf) => {
  proxyMesh = gltf.scene;
  proxyMesh.visible = false; // invisible; splat is visual
  scene.add(proxyMesh);
});
2 — raycast on tap → 3D hit point
canvas.addEventListener('click', (e) => {
  pointer.x = (e.clientX / W) * 2 - 1;
  pointer.y = -(e.clientY / H) * 2 + 1;
  raycaster.setFromCamera(pointer, camera);
  const hits = raycaster
    .intersectObject(proxyMesh, true);
  if (hits.length) {
    pendingPos = hits[0].point; // Vector3
    openSheet();
  }
});

// Spark.js shader uniform

// GLSL fragment (simplified)

4 — store to Supabase PostGIS
await supabase
  .from('pins')
  .insert({
    user_id:    deviceHash(),
    condition:  '3D',
    value_type: selectedVal,
    note:       noteText,
    scene_x: pos.x, scene_y: pos.y,
    scene_z: pos.z,
    dwell_ms: Date.now() - loadTime,
    ts: new Date().toISOString()
  });
